Ví dụ: Bạn có thể tạo một file /etc/wireplumber/main.lua.d/60-custom-policy.lua (hoặc trong ~/.config/wireplumber/...) 
với nội dung tương tự như sau để đặt ưu tiên cho các loại thiết bị:

rule = {
  matches = {
    {
      -- Match all audio sinks
      { "node.name", "matches", "alsa_output.*" },
    },
  },
  apply_properties = {
    -- Default priority
    ["priority.session"] = 100,
  },
}
table.insert(alsa_monitor.rules, rule)

rule = {
  matches = {
    {
      -- Match 3.5mm jack (card 1 on RPi)
      { "device.name", "matches", "alsa_card.platform-bcm2835_headpho" },
    },
  },
  apply_properties = {
    -- Higher priority than default
    ["priority.session"] = 1500,
  },
}
table.insert(alsa_monitor.rules, rule)

rule = {
  matches = {
    {
      -- Match all bluetooth devices
      { "device.name", "matches", "bluez_card.*" },
    },
  },
  apply_properties = {
    -- Highest priority
    ["priority.session"] = 2000,
  },
}
table.insert(bluez_monitor.rules, rule)
